\hypertarget{usage}{%
\subsection{Usage}\label{usage}}

Place \texttt{.md} files here with YAML frontmatter:

\begin{Shaded}
\begin{Highlighting}[]
\PreprocessorTok{{-}{-}{-}}
\FunctionTok{title}\KeywordTok{:}\AttributeTok{ }\StringTok{"Document Title"}
\FunctionTok{author}\KeywordTok{:}\AttributeTok{ }\StringTok{"Author Name"}
\FunctionTok{date}\KeywordTok{:}\AttributeTok{ }\StringTok{"December 2025"}
\FunctionTok{lang}\KeywordTok{:}\AttributeTok{ en}
\FunctionTok{chapterstyle}\KeywordTok{:}\AttributeTok{ minimal}
\PreprocessorTok{{-}{-}{-}}

\CommentTok{\# Your Content}

\AttributeTok{Write your document in markdown...}
\end{Highlighting}
\end{Shaded}

\hypertarget{build}{%
\subsection{Build}\label{build}}

\begin{Shaded}
\begin{Highlighting}[]
\CommentTok{\# Build all documents}
\ExtensionTok{./Veritas.Engine/Script/build.sh}

\CommentTok{\# Build specific document}
\ExtensionTok{./Veritas.Engine/Script/build.sh}\NormalTok{ document.md}
\end{Highlighting}
\end{Shaded}

Output PDFs appear in \texttt{Veritas.Output/}.
