\newglossaryentry{multidisciplinary}
{
    name=multidisciplinary care,
    plural=multidisciplinary care approaches,
    description={A healthcare approach involving multiple healthcare professionals from different disciplines working together in a coordinated manner to provide comprehensive patient care}
}

\newglossaryentry{burnunit}
{
    name=burn unit,
    plural=burn units,
    description={A specialized healthcare facility equipped with dedicated resources, staff expertise, and protocols specifically designed for the comprehensive treatment of burn injuries}
}

\newglossaryentry{parkland}
{
    name=Parkland Formula,
    description={A formula for calculating fluid resuscitation requirements in burn patients: 4mL/kg/\% TBSA of crystalloid over 24 hours}
}

\newglossaryentry{bauxscore}
{
    name=Baux score,
    description={A prognostic score for burn mortality calculated as age plus percent TBSA burned. Scores above 140 historically associated with greater than 90\% mortality}
}

\newglossaryentry{culturalsafety}
{
    name=cultural safety,
    description={An approach to healthcare that recognizes and respects cultural diversity, addresses power imbalances, and enables culturally appropriate service provision}
}

\newglossaryentry{telehealth}
{
    name=telehealth,
    description={The delivery of healthcare services remotely using telecommunications technology, enabling consultation and treatment without physical presence}
}

\newglossaryentry{sepsis}
{
    name=sepsis,
    description={Life-threatening organ dysfunction caused by a dysregulated host response to infection, defined by Sepsis-3 criteria}
}

\newglossaryentry{contracture}
{
    name=contracture,
    plural=contractures,
    description={A permanent shortening of muscle or joint tissue that restricts movement, commonly occurring as a complication of burn injuries}
}
