\newglossaryentry{multidisciplinary}
{
    name=Multidisciplinary Care,
    description={A healthcare approach involving multiple healthcare professionals from different disciplines working together in a coordinated manner to provide comprehensive patient care. In burn care, this typically includes surgeons, nurses, physiotherapists, occupational therapists, dietitians, psychologists, and social workers}
}

\newglossaryentry{burnunit}
{
    name=Specialist Burn Unit,
    description={A specialized healthcare facility equipped with dedicated resources, staff expertise, and protocols specifically designed for the comprehensive treatment of burn injuries. These units typically handle complex and severe burn cases requiring intensive multidisciplinary care}
}

\newglossaryentry{evidencesynthesis}
{
    name=Evidence Synthesis,
    description={The process of systematically reviewing, analyzing, and combining evidence from multiple research studies to draw comprehensive conclusions about a specific research question or clinical practice area}
}

\newglossaryentry{criticalappraisal}
{
    name=Critical Appraisal,
    description={The systematic evaluation of research studies to assess their validity, reliability, and applicability to clinical practice. This involves examining study design, methodology, results, and potential biases}
}

\newglossaryentry{functionaloutcome}
{
    name=Functional Outcome,
    description={Measures of a patient's ability to perform activities of daily living and return to pre-injury levels of physical, social, and occupational functioning following burn injury and treatment}
}
