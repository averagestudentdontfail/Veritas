\newglossaryentry{anzba}
{
    name=ANZBA,
    description={Australian and New Zealand Burn Association: Professional organisation coordinating burn care standards, education, and research across Australasia.}
}

\newglossaryentry{branz}
{
    name=BRANZ,
    description={Burns Registry of Australia and New Zealand: Comprehensive clinical registry collecting standardised data from all major burn centres in Australia and New Zealand.}
}

\newglossyentry{chronic pain}
{
    name=Chronic Pain,
    description={Persistent pain lasting longer than 3 months, or beyond normal tissue healing time, involving complex neurophysiological and psychological mechanisms.}
}

\newglossaryentry{cultural safety}
{
    name=Cultural Safety,
    description={Healthcare practice acknowledging power imbalances and working to address discrimination, achieving equitable health outcomes for Indigenous and culturally diverse populations.}
}

\newglossaryentry{interdisplinary care}
{
    name=Interdisciplinary Care,
    description={Coordinated healthcare delivery involving multiple professional disciplines working collaboratively with shared goals, communication protocols, and care planning.}
}

\newglossaryentry{multimodal analgesia}
{
    name=Multimodal Analgesia,
    description={Pain management approach combining multiple analgesic mechanisms, both pharmacological and non-pharmacological, to optimise pain relief whilst minimising adverse effects.}
}

\newglossaryentry{patient-reported measures}
{
    name=Patient-Reported Measures (PRMs),
    description={ Standardised instruments capturing patients' direct reports of their health status, symptoms, and quality of life without interpretation by healthcare providers.}
}

\newglossaryentry{total body surface area}
{
    name=Total Body Surface Area (TBSA),
    description={Percentage of body surface affected by burn injury, calculated using standardised assessment methods (\textit{Rule of Nines} or \textit{Lund-Browder charts}).}
}

\newglossaryentry{trauma-informed care}
{
    name=Trauma-Informed Care,
    description={Healthcare approach recognising the impact of trauma on individuals and systems, emphasising safety, trustworthiness, peer support, collaboration, empowerment, and cultural humility.}
}

\newglossaryentry{virtual care}
{
    name=Virtual Care,
    description={Healthcare delivery using technology to enable remote consultation, monitoring, and intervention between patients and healthcare providers or between healthcare providers.}
}
