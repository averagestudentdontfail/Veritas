\begin{tikzpicture}[remember picture, overlay]
    % White-gray background like photocopied paper
    \fill[gray!8!white] (current page.south west) rectangle (current page.north east);
    
    % Add photocopy artifacts/noise
    \foreach \i in {1,...,200}{
        \pgfmathsetmacro\xpos{rand*\paperwidth}
        \pgfmathsetmacro\ypos{rand*\paperheight}
        \pgfmathsetmacro\size{rand*0.3+0.1}
        \pgfmathsetmacro\opac{rand*0.08+0.02}
        \fill[black, opacity=\opac] 
            ($(current page.south west) + (\xpos pt, \ypos pt)$) 
            circle (\size mm);
    }
    
    % Add vertical scan lines (subtle)
    \foreach \x in {1,3,...,40}{
        \pgfmathsetmacro\opac{rand*0.02+0.01}

            ($(current page.north west) + (\x cm, 0)$) -- 
            ($(current page.south west) + (\x cm, 0)$);
    }
    
    % Add slight shadow on left edge (from scanner lid)
    \shade[left color=black!15!white, right color=white, opacity=0.3] 
        (current page.south west) rectangle ($(current page.north west) + (0.5cm, 0)$);
        
    % Add top shadow (scanner edge)
    \shade[top color=black!10!white, bottom color=white, opacity=0.2] 
        (current page.north west) rectangle ($(current page.north east) + (0, -1cm)$);
\end{tikzpicture}
